function getPosts() { try { return JSON.parse(localStorage.getItem("posts")) || []; } catch (e) { return []; } } function savePosts(posts) { localStorage.setItem("posts", JSON.stringify(posts)); } function addPost() { const input = document.getElementById("postInput"); const text = input.value.trim(); if (!text) return; let posts = getPosts(); posts.unshift({ user: "Invité", text: text, time: new Date().toLocaleString() }); savePosts(posts); input.value = ""; renderPosts(); } function renderPosts() { const feed = document.getElementById("feed"); feed.innerHTML = ""; let posts = getPosts(); posts.forEach(post => { const div = document.createElement("div"); div.className = "post"; div.innerHTML = ` ${post.user}

${post.text}

${post.time} `; feed.appendChild(div); }); } window.onload = renderPosts;
