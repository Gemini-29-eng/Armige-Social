function addPost() { const input = document.getElementById("postInput"); const text = input.value; if (!text) return; let posts = JSON.parse(localStorage.getItem("posts")) || []; posts.unshift({ text: text, time: new Date().toLocaleString() }); localStorage.setItem("posts", JSON.stringify(posts)); input.value = ""; renderPosts(); } function renderPosts() { const feed = document.getElementById("feed"); feed.innerHTML = ""; let posts = JSON.parse(localStorage.getItem("posts")) || []; posts.forEach(post => { const div = document.createElement("div"); div.className = "post"; div.innerHTML = `

${post.text}

${post.time} `; feed.appendChild(div); }); } window.onload = renderPosts;
